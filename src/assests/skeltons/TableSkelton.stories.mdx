# Table Skelton

Display a Skelton when you are fetching async table

## Example
```jsx
import React, { FC, useState, useEffect } from 'react';
import TableSkelton from 'assets/skeltons/ListingSkelton.component';

const MyComponent: FC<{}> = () => {
    const [data, setData] = useState();
    
    useEffect(() => {
        fetch('https://some-url/endpoint')
        .then(res => res.json())
        .then(res => setData(res.data));
    }, []);

    return (
        <TableSkelton /> && <p>Hi {data.name}</p>
    );
}
```
