# Listing Skelton

Display a Skelton when you are fetching async listings

## Example
```jsx
import React, { FC, useState, useEffect } from 'react';
import ListingSkelton from 'assets/skeltons/ListingSkelton.component';

const MyComponent: FC<{}> = () => {
    const [data, setData] = useState();
    
    useEffect(() => {
        fetch('https://some-url/endpoint')
        .then(res => res.json())
        .then(res => setData(res.data));
    }, []);

    return (
        <ListingSkelton /> && <p>Hi {data.name}</p>
    );
}
```
